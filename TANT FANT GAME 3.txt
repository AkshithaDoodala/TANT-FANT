int check_win()
{
    if(pos[0]==pos[1]&&pos[1]==pos[2])
        return 1;
    else if(pos[3]==pos[4]&&pos[4]==pos[5])        //Horizontal winning condition

    else if(pos[1]==pos[4]&&pos[4]==pos[7])        //Vertical winning condition

}
